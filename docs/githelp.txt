# git commands

# pull updates from the repo
git pull

# make some changes

# add untracked files
git add -A

# commit your changes
git commit -m '<commit message>'

# push updates to the repo
git push

# if 'git pull' results in conflict and you have made changes, store them temporarily
git stash

# then pull updates and restore your changes back
git stash pop

# to go back to specific commit
git reset <commit SHA id>
